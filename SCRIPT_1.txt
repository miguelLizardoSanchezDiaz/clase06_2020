CREATE TABLE categoria (id int(10) NOT NULL AUTO_INCREMENT, cat_codigo varchar(50), cat_descripcion varchar(200), cat_estado int(10), PRIMARY KEY (id));

CREATE TABLE detalle_comprobante (id int(11) NOT NULL AUTO_INCREMENT, com_id int(11) NOT NULL, pro_id int(11) NOT NULL, dco_cantidad decimal(10, 4), dco_precio decimal(10, 2), dco_importe decimal(10, 2), dco_descuento decimal(10, 2), dco_total decimal(10, 2), dco_estado int(11), PRIMARY KEY (id));

CREATE TABLE marca (id int(11) NOT NULL AUTO_INCREMENT, mar_codigo varchar(50), mar_descripcion varchar(200), mar_estado int(11), PRIMARY KEY (id));

CREATE TABLE producto (id int(11) NOT NULL AUTO_INCREMENT, pro_codigo varchar(50), pro_descripcion varchar(250), pro_fecha_creacion date, pro_ruta_foto varchar(280), pro_tipo_producto int(11), pro_peso decimal(10, 2), pro_volumen decimal(10, 2), pro_longitud decimal(10, 2), pro_estado int(11), pro_caracteristicas varchar(250), cat_id int(10) NOT NULL, unm_id int(10) NOT NULL, mar_id int(11) NOT NULL, mod_id int(11) NOT NULL, PRIMARY KEY (id));
